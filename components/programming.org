** eglot helper
Shared by language components.  Starts the built-in LSP client only when a
matching language server is actually installed, so enabling a component never
breaks a buffer just because the server is missing.

#+BEGIN_SRC emacs-lisp
(defun kwarks/eglot-ensure-if (&rest programs)
  "Run `eglot-ensure' when one of PROGRAMS is found on the PATH."
  (when (seq-some #'executable-find programs)
    (require 'eglot)
    (eglot-ensure)))
#+END_SRC

** diff-hl
Show VCS change indicators in the fringe (replaces git-gutter).

#+BEGIN_SRC emacs-lisp
(use-package diff-hl
  :demand t
  :hook ((dired-mode . diff-hl-dired-mode)
         (magit-pre-refresh . diff-hl-magit-pre-refresh)
         (magit-post-refresh . diff-hl-magit-post-refresh))
  :config
  (global-diff-hl-mode 1)
  (when (require 'diff-hl-flydiff nil t)
    (diff-hl-flydiff-mode 1))

  (defhydra hydra-diff-hl (:hint nil)
    "
Diff-hl:
  _j_: next hunk      _s_tage hunk     _q_uit
  _k_: previous hunk  _r_evert hunk
  ^ ^                 _p_opup/show hunk
"
    ("j" diff-hl-next-hunk)
    ("k" diff-hl-previous-hunk)
    ("s" diff-hl-stage-current-hunk)
    ("r" diff-hl-revert-hunk)
    ("p" diff-hl-show-hunk)
    ("q" nil :color blue))

  (defun git-gutter-actions ()
    "Backwards-compatible entry point to the diff-hl hydra."
    (interactive)
    (hydra-diff-hl/body)))
#+END_SRC
