** Tree-sitter (Emacs 31)
Emacs 31 can auto-remap supported major modes to their tree-sitter variants when
grammars are installed, replacing manual =*-ts-mode= wiring and third-party
helpers.  Guarded so it only runs on Emacs 31+ with a working tree-sitter
library.  Grammars are never auto-downloaded from Emacs.  The command
=treesit-install-language-grammar= installs **one** grammar per run.  For
several at once with the same minibuffer filtering as =M-x=, use
=kwarks/treesit-install-languages= (comma-separated symbols, TAB completes each
element).  For English / Russian / Spanish spell-checking see [[file:jinx.org][jinx.org]]
(=jinx-languages=).

#+BEGIN_SRC emacs-lisp
(when (and (>= emacs-major-version 31)
           (fboundp 'treesit-available-p)
           (treesit-available-p))
  (require 'treesit)
  ;; `setopt' so the :set logic updates `major-mode-remap-alist'.
  (setopt treesit-enabled-modes t
          ;; Never auto-download grammars from Emacs (install via OS or `treesit-install-language-grammar').
          treesit-auto-install-grammar 'never))
#+END_SRC

** Tree-sitter grammar batch install
=treesit-install-language-grammar= only reads **one** language per invocation.
This command uses =completing-read-multiple= (from =crm.el=): type several names
separated by commas; completion and filtering apply to **each** element, so
Vertico behaves like familiar =M-x= completion.  It only requires a Tree-sitter
capable Emacs (=treesit-available-p=), not necessarily Emacs 31.  Requires Git
and a C/C++ toolchain on =exec-path=.

Emacs 30 leaves =treesit-language-source-alist= nil by default; this command
merges =kwarks/treesit-default-grammar-recipes= for common languages only when
a language has no recipe yet (your own =setq=/Customize entries win).  For any
other grammar, add a recipe to =treesit-language-source-alist= first (see Info
node =(elisp) Parser-based Font Lock= and =treesit-install-language-grammar=).
Languages whose grammar already loads (=treesit-language-available-p=) are
omitted from completion so you only pick what still needs installing.  While
typing comma-separated names, languages already chosen in earlier segments are
hidden from the candidate list for the current segment.

#+BEGIN_SRC emacs-lisp
(defun kwarks/treesit--grammar-loadable-p (lang)
  "Return non-nil if Tree-sitter can already load the grammar for LANG."
  (and (fboundp 'treesit-language-available-p)
       (car (treesit-language-available-p lang t))))

(defun kwarks/treesit--upgrade-obsolete-recipe-urls ()
  "Fix defunct grammar URLs that were never under github.com/tree-sitter/.
Safe to run repeatedly; only replaces known-bad URL strings."
  (let ((old->new
         '(("https://github.com/tree-sitter/tree-sitter-yaml" . "https://github.com/tree-sitter-grammars/tree-sitter-yaml")
           ("https://github.com/tree-sitter/tree-sitter-yaml.git" . "https://github.com/tree-sitter-grammars/tree-sitter-yaml")
           ("https://github.com/tree-sitter/tree-sitter-lua" . "https://github.com/tree-sitter-grammars/tree-sitter-lua")
           ("https://github.com/tree-sitter/tree-sitter-lua.git" . "https://github.com/tree-sitter-grammars/tree-sitter-lua")
           ("https://github.com/tree-sitter/tree-sitter-r" . "https://github.com/r-lib/tree-sitter-r")
           ("https://github.com/tree-sitter/tree-sitter-r.git" . "https://github.com/r-lib/tree-sitter-r"))))
    (dolist (cell treesit-language-source-alist)
      (let* ((recipe (cdr cell))
             (url (car-safe recipe)))
        (when (stringp url)
          (when-let ((new (cdr (assoc url old->new #'string=))))
            (setcar recipe new)))))))

(defun kwarks/treesit--merge-default-grammar-recipes ()
  "Add `kwarks/treesit-default-grammar-recipes' entries missing from the alist.
Emacs 30 ships `treesit-language-source-alist' as nil; major modes do not
populate it globally, so batch install needs a baseline list."
  (kwarks/treesit--upgrade-obsolete-recipe-urls)
  (dolist (cell kwarks/treesit-default-grammar-recipes)
    (let ((lang (car cell)))
      (unless (assq lang treesit-language-source-alist)
        (push (cons lang (cdr cell)) treesit-language-source-alist)))))

(defun kwarks/treesit--crm-completion-table (candidates)
  "Return a programmed completion table for string CANDIDATES.
For `completing-read-multiple', drops names that appear in earlier
comma-separated fields (text before the current field per `crm--current-element')."
  (lambda (string pred action)
    (let* ((already (when (and (minibufferp) (fboundp 'crm--current-element))
                      (let* ((bounds (crm--current-element))
                             (prefix (buffer-substring-no-properties (minibuffer-prompt-end)
                                                                       (car bounds))))
                        (mapcar #'string-trim
                                (split-string prefix crm-separator t)))))
           (table (seq-remove (lambda (name) (member name already)) candidates)))
      (complete-with-action action table string pred))))

(defun kwarks/treesit-install-languages ()
  "Install several Tree-sitter grammars via `completing-read-multiple'.
Each name must exist in `treesit-language-source-alist' after defaults merge.
Languages that already load (`treesit-language-available-p') are not offered.
While entering comma-separated names, languages already picked in earlier
fields are omitted from completion for the current field."
  (interactive)
  (unless (and (fboundp 'treesit-available-p)
               (treesit-available-p))
    (user-error
     "Tree-sitter is not available in this session (need Emacs built with libtree-sitter; check `treesit-available-p')"))
  (require 'treesit)
  (kwarks/treesit--merge-default-grammar-recipes)
  (require 'crm)
  (let* ((recipe-langs (mapcar #'car treesit-language-source-alist))
         (pending (delq nil (mapcar (lambda (lang)
                                      (unless (kwarks/treesit--grammar-loadable-p lang)
                                        lang))
                                    recipe-langs)))
         (names (sort (mapcar #'symbol-name pending) #'string<)))
    (unless names
      (user-error "No grammars left to install (every recipe language already loads, or no recipes)"))
    (let ((picked (completing-read-multiple
                   (format "%d/%d Tree-sitter languages (comma-separated, TAB completes each): "
                           (length names) (length recipe-langs))
                   (kwarks/treesit--crm-completion-table names)
                   nil t)))
      (dolist (s picked)
        (let ((lang (intern s)))
          (unless (memq lang pending)
            (error "Unknown or already-installed language `%s'" lang))
          (message "Installing Tree-sitter grammar `%s'..." lang)
          (condition-case err
              (treesit-install-language-grammar lang)
            (error (message "`%s' failed: %S" lang err))))))))
#+END_SRC

** eglot helper
Shared by language components.  Starts the built-in LSP client only when a
matching language server is actually installed, so enabling a component never
breaks a buffer just because the server is missing.

#+BEGIN_SRC emacs-lisp
(defun kwarks/eglot-ensure-if (&rest programs)
  "Run `eglot-ensure' when one of PROGRAMS is found on the PATH."
  (when (seq-some #'executable-find programs)
    (require 'eglot)
    (eglot-ensure)))
#+END_SRC

** diff-hl
Show VCS change indicators in the fringe (replaces git-gutter).

#+BEGIN_SRC emacs-lisp
(use-package diff-hl
  :demand t
  :init
  ;; `define-globalized-minor-mode' binds this only after :autoload-end in
  ;; diff-hl.el; hooks or mode-line work (e.g. Marginalia during `C-x b') can run
  ;; `global-diff-hl-mode-enable-in-buffer' first and signal void-variable after
  ;; switching Emacs versions or with a narrow autoload load.
  (defvar global-diff-hl-mode-buffers nil)
  :hook ((dired-mode . diff-hl-dired-mode)
         (magit-pre-refresh . diff-hl-magit-pre-refresh)
         (magit-post-refresh . diff-hl-magit-post-refresh))
  :config
  (global-diff-hl-mode 1)
  (when (require 'diff-hl-flydiff nil t)
    (diff-hl-flydiff-mode 1))

  (defhydra hydra-diff-hl (:hint nil)
    "
Diff-hl:
  _j_: next hunk      _s_tage hunk     _q_uit
  _k_: previous hunk  _r_evert hunk
  ^ ^                 _p_opup/show hunk
"
    ("j" diff-hl-next-hunk)
    ("k" diff-hl-previous-hunk)
    ("s" diff-hl-stage-current-hunk)
    ("r" diff-hl-revert-hunk)
    ("p" diff-hl-show-hunk)
    ("q" nil :color blue))

  (defun git-gutter-actions ()
    "Backwards-compatible entry point to the diff-hl hydra."
    (interactive)
    (hydra-diff-hl/body)))
#+END_SRC
