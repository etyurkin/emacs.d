** exec-path-from-shell
Ensure environment variables inside Emacs look the same as in the user's shell.

Use a non-interactive login shell (=-l= without =-i=) to import the environment;
this is much faster than spawning a full interactive shell.  Note: this reads
=.zprofile/.zshenv= but not =.zshrc=, so any =PATH=/env you rely on must live in
the former.  If something goes missing, restore the default by setting
=exec-path-from-shell-arguments= to ='("-l" "-i")=.

#+BEGIN_SRC emacs-lisp
(use-package exec-path-from-shell
  :demand t
  :init
  (setq exec-path-from-shell-arguments '("-l"))
  :config
  (dolist (var '("SSH_AUTH_SOCK" "SSH_AGENT_PID" "GPG_AGENT_INFO" "LANG" "LC_CTYPE"))
    (add-to-list 'exec-path-from-shell-variables var))
  (when (memq window-system '(mac ns x))
    (exec-path-from-shell-initialize)))
#+END_SRC

** auto-dim-other-buffers
Make the non-current buffer less prominent.

#+BEGIN_SRC emacs-lisp
;; How much to darken the inactive (dimmed) buffer background, in HSL lightness
;; points.  Bump these up for more contrast between active and inactive buffers.
(defvar kwarks/auto-dim-darken-dark 12
  "Darken amount for the dimmed face on dark themes.")
(defvar kwarks/auto-dim-darken-light 6
  "Darken amount for the dimmed face on light themes.")

(defun kwarks/set-auto-dim-other-buffers-background (&rest args)
  "Make the inactive (dimmed) buffer background darker than the active one."
  (let* ((bg (face-background 'default))
         (bg (if (or (null bg) (string= bg "unspecified-bg")) "#000000" bg)))
    (set-face-background
     'auto-dim-other-buffers-face
     (color-darken-name bg (kwarks/if-dark-theme
                            kwarks/auto-dim-darken-dark
                            kwarks/auto-dim-darken-light)))))

(use-package auto-dim-other-buffers
  :config
  (setq auto-dim-other-buffers-dim-on-focus-out nil)
  (kwarks/set-auto-dim-other-buffers-background)
  :init
  (auto-dim-other-buffers-mode t))

(advice-add 'load-theme :after 'kwarks/set-auto-dim-other-buffers-background)
#+END_SRC

** doom-modeline
#+BEGIN_SRC emacs-lisp
(defun init-doom-modeline ()
  (require 'doom-modeline)

  (setq doom-modeline-icon nil)
  (doom-modeline-mode)

  (setq display-time-day-and-date t)
  (setq display-time-string-forms
        '((propertize (format-time-string " %a %b %d, %R" now) 'weight 'normal)))
  (display-time-mode t)

  (doom-modeline-def-modeline 'my-simple-line
    '(matches buffer-info remote-host parrot)
    '(buffer-position selection-info minor-modes process vcs input-method buffer-encoding battery misc-info major-mode))

  (doom-modeline-set-modeline 'my-simple-line 'default))

(use-package doom-modeline
  :hook (after-init . init-doom-modeline))
#+END_SRC

** battery
Show battery level in the mode line when a battery is present (built-in).

#+BEGIN_SRC emacs-lisp
(defun has-battery ()
  (require 'battery)
  (if battery-status-function
      (let ((b-status (cdr (assoc '66 (funcall battery-status-function)))))
	(not (string= b-status "N/A")))))

(when (has-battery)
  (setq battery-mode-line-format " [%b%p%%]")
  (display-battery-mode 1))
#+END_SRC

** jinx
Fast spell checking using the Enchant library.  On first activation jinx
compiles a small native module (=jinx-mod.c=) that links against =enchant-2=,
so the following must be present at build time:
- =enchant= and =pkg-config= (=brew install enchant pkg-config=),
- at least one dictionary backend reachable by Enchant (=aspell= or =hunspell=,
  both pulled in as Homebrew deps of =enchant=).

Use =M-x jinx-languages= to inspect/select the available dictionaries.
Activation is guarded below so a missing native library never breaks editing.

#+BEGIN_SRC emacs-lisp
(use-package jinx
  :bind (("M-$"   . jinx-correct)
         ("C-M-$" . jinx-languages))
  :init
  (defun kwarks/maybe-jinx ()
    "Enable jinx-mode if the enchant native module is available."
    (when (ignore-errors (require 'jinx nil t))
      (ignore-errors (jinx-mode 1))))
  (add-hook 'text-mode-hook #'kwarks/maybe-jinx)
  (add-hook 'org-mode-hook  #'kwarks/maybe-jinx))
#+END_SRC
