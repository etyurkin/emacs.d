** Theme helper functions
#+BEGIN_SRC emacs-lisp
(require 'color)

(defun kwarks/usable-color-p (color)
  "Non-nil when COLOR is a string Emacs can turn into RGB."
  (and (stringp color)
       (not (member color '("unspecified-bg" "unspecified-fg" "unspecified")))
       (color-name-to-rgb color)
       color))

(defun kwarks/default-background ()
  "Return the default background as a usable color string.
Tries the default face (resolving inheritance) and then the frame's
`background-color' parameter, because on some Emacs versions (e.g.
emacs-plus@31) `face-background' returns \"unspecified-bg\" while theme
colors are being applied.  Falls back to black only as a last resort."
  (or (kwarks/usable-color-p (face-background 'default nil t))
      (kwarks/usable-color-p (frame-parameter nil 'background-color))
      "#000000"))

(defun kwarks/dark-theme-p ()
  "Return non-nil when the default background is dark.
Decided from the background's actual luminance rather than the frame's
`background-mode' parameter, which is not reliably set on all Emacs
versions (e.g. emacs-plus@31) when theme colors are applied."
  (let ((rgb (color-name-to-rgb (kwarks/default-background))))
    ;; RGB components are 0..1; their sum is 0..3, so < 1.5 is dark.
    (and rgb (< (apply #'+ rgb) 1.5))))

(defmacro kwarks/if-dark-theme (action else-action)
  `(if (kwarks/dark-theme-p)
       ,action
     ,else-action))

(defun kwarks/color-from-theme (lighter-percentage darker-percentage)
  (let ((bg-color (kwarks/default-background)))
    (kwarks/if-dark-theme
     (color-lighten-name bg-color lighter-percentage)
     (color-darken-name bg-color darker-percentage))))
#+END_SRC

#+BEGIN_SRC emacs-lisp
(defun visual-line-range ()
  (save-excursion
    (cons
     (progn (vertical-motion 0) (point))
     (progn (vertical-motion 1) (point)))))

(setq hl-line-range-function 'visual-line-range)

(global-hl-line-mode 1)

** Main theme
#+BEGIN_SRC emacs-lisp
(defmacro kwarks/with-make-frame (&rest body)
  `(if (daemonp)
       (add-hook 'after-make-frame-functions
                 (lambda (&rest args) (progn ,@body)) t)
     ,@body))

(defun kwarks/set-cursor-color ()
  (set-face-background 'cursor (kwarks/color-from-theme 100 100)))

(defun kwarks/set-hl-line-color ()
  (set-face-background 'hl-line (kwarks/color-from-theme 6 7)))

(defun kwarks/set-region-color ()
  (set-face-background 'region (color-lighten-name "#2D383E" 10)))

(defun kwarks/set-border-color ()
  (let ((color (kwarks/color-from-theme 5 10)))
    (set-face-attribute 'mode-line nil :foreground "gray" :background 'unspecified :box nil :overline color)
    (set-face-attribute 'mode-line-inactive nil :background 'unspecified :box nil :foreground "gray" :overline color)
    (set-face-attribute 'vertical-border nil :background 'unspecified :foreground color)))

(defun kwarks/set-titlebar-style ()
  (set-frame-parameter nil 'ns-transparent-titlebar t)
  (kwarks/if-dark-theme
   (set-frame-parameter nil 'ns-appearance 'dark)
   (set-frame-parameter nil 'ns-appearance nil)))

(defun kwarks/set-theme-colors (&rest args)
  (kwarks/set-cursor-color)
  (kwarks/set-hl-line-color)
  (kwarks/set-region-color)
  (kwarks/set-border-color)
  (kwarks/set-titlebar-style)
  (set-frame-parameter nil 'alpha '(90 . 90)))

(defun kwarks/disable-other-themes (&rest _)
  "Disable already-enabled themes before loading a new one."
  (mapc #'disable-theme custom-enabled-themes))

(advice-add 'load-theme :before #'kwarks/disable-other-themes)
(advice-add 'load-theme :after #'kwarks/set-theme-colors)

(defun kwarks/set-theme-faces ()
  (custom-set-faces
   `(cursor                   ((t (:background ,(kwarks/color-from-theme 100 100)))))
   '(font-lock-comment-face   ((t (:slant italic :background unspecified))))
   '(trailing-whitespace      ((t (:foreground "gray"     :background "gray"))))
   '(fringe                   ((t (:foreground "SkyBlue4" :background unspecified))))))

(defun kwarks/setup-frame (&rest args)
  "Apply titlebar theme for new frames"
  (kwarks/set-titlebar-style)
  (kwarks/set-theme-colors)
  (kwarks/set-theme-faces))

(kwarks/with-make-frame (kwarks/setup-frame))
(add-hook 'after-make-frame-functions 'kwarks/setup-frame t)

(use-package doom-themes
  :init
  (setq doom-one-brighter-comments t)
  ;; Break a face-inheritance cycle that Emacs 30+ flags loudly: Gnus defines
  ;; `gnus-group-news-low' to inherit `gnus-group-news-low-empty', while
  ;; doom-themes makes the -empty face inherit the non-empty one -- a loop.
  ;; Pinning a top-priority user spec for both (before `load-theme') gives a
  ;; clean, non-cyclic chain so the warning never fires.
  (custom-set-faces
   '(gnus-group-news-low ((t (:inherit gnus-group-mail-1))))
   '(gnus-group-news-low-empty ((t (:inherit gnus-group-mail-1-empty)))))
  (load-theme 'doom-one t))
#+END_SRC

** Theme cache
Write the active theme's colors and font to =theme-cache.el= so =early-init.el=
can apply them to =default-frame-alist= before the frame is created on the next
start.  This eliminates the white flash and the small-window-then-resize dance
(the latter is caused by the font being applied after the frame was sized).

#+BEGIN_SRC emacs-lisp
(defun kwarks/save-theme-cache ()
  "Write current theme colors and font to theme-cache.el.
These are pushed into `default-frame-alist' by early-init.el before the
frame exists, so the window starts with the right colors AND font.
Caching the font is what prevents the late resize: desktop restores the
frame size in character units, so the frame must already use the final
font when those characters are mapped to pixels, otherwise applying the
font afterwards re-maps the grid and the frame shrinks.
Frame geometry itself is left to desktop-restore which already handles it."
  (when (display-graphic-p)
    (let* ((cache (expand-file-name "theme-cache.el" kwarks/cache-directory))
           (frame (selected-frame))
           (bg (face-background 'default frame))
           (fg (face-foreground 'default frame))
           (font (frame-parameter frame 'font))
           (dark-p (eq (frame-parameter frame 'background-mode) 'dark)))
      (with-temp-file cache
        (insert ";;; theme-cache.el --- auto-generated, do not edit -*- no-byte-compile: t; lexical-binding: t; -*-\n")
        (when bg (insert (format "(push '(background-color . %S) default-frame-alist)\n" bg)))
        (when fg (insert (format "(push '(foreground-color . %S) default-frame-alist)\n" fg)))
        (when (stringp font)
          (insert (format "(push '(font . %S) default-frame-alist)\n" font)))
        (when dark-p
          (insert "(push '(ns-appearance . dark) default-frame-alist)\n"))))))

(advice-add 'load-theme :after (lambda (&rest _) (kwarks/save-theme-cache)))
(add-hook 'kill-emacs-hook #'kwarks/save-theme-cache)
#+END_SRC

    '(org-block-end-line      ((t (:background unspecified :slant italic))))
    '(font-lock-comment-face  ((t (:background unspecified))))
    '(org-ellipsis            ((t (:background unspecified :strike-through nil))))
    '(org-done                ((t (:strike-through t))))
    '(org-headline-done       ((t (:strike-through t))))
