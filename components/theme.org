** Theme helper functions
#+BEGIN_SRC emacs-lisp
(require 'color)

(defmacro kwarks/if-dark-theme (action else-action)
  `(if (eq (frame-parameter nil 'background-mode) 'dark)
       ,action
     ,else-action))

(defun kwarks/color-from-theme (lighter-percentage darker-percentage)
  (let* ((bg (face-background 'default))
         (bg-color (if (string= bg "unspecified-bg") "#000000" bg)))

    (kwarks/if-dark-theme
     (color-lighten-name bg-color lighter-percentage)
     (color-darken-name bg-color darker-percentage))))
#+END_SRC

** Highlight current line
Ignore line wrappings and highlight entire line for org headers

#+BEGIN_SRC emacs-lisp
(defun visual-line-range ()
  (save-excursion
    (cons
     (progn (vertical-motion 0) (point))
     (progn (vertical-motion 1) (point)))))

(setq hl-line-range-function 'visual-line-range)

(global-hl-line-mode 1)
#+END_SRC

** Main theme
#+BEGIN_SRC emacs-lisp
(defmacro kwarks/with-make-frame (&rest body)
  `(if (daemonp)
       (add-hook 'after-make-frame-functions
                 (lambda (&rest args) (progn ,@body)) t)
     ,@body))

(defun kwarks/set-cursor-color ()
  (set-face-background 'cursor (kwarks/color-from-theme 100 100)))

(defun kwarks/set-hl-line-color ()
  (set-face-background 'hl-line (kwarks/color-from-theme 6 7)))

(defun kwarks/set-region-color ()
  (set-face-background 'region (color-lighten-name "#2D383E" 10)))

(defun kwarks/set-border-color ()
  (let ((color (kwarks/color-from-theme 5 10)))
    (set-face-attribute 'mode-line nil :foreground "gray" :background 'unspecified :box nil :overline color)
    (set-face-attribute 'mode-line-inactive nil :background 'unspecified :box nil :foreground "gray" :overline color)
    (set-face-attribute 'vertical-border nil :background 'unspecified :foreground color)))

(defun kwarks/set-titlebar-style ()
  (set-frame-parameter nil 'ns-transparent-titlebar t)
  (kwarks/if-dark-theme
   (set-frame-parameter nil 'ns-appearance 'dark)
   (set-frame-parameter nil 'ns-appearance nil)))

(defun kwarks/set-theme-colors (&rest args)
  (kwarks/set-cursor-color)
  (kwarks/set-hl-line-color)
  (kwarks/set-region-color)
  (kwarks/set-border-color)
  (kwarks/set-titlebar-style)
  (set-frame-parameter nil 'alpha '(90 . 90)))

(defun kwarks/set-theme-font ()
  "Use JetBrains Mono font when available. The font should be installed from .emacs.d/fonts folder"
  (let ((theme-font "JetBrains Mono"))
    (when (member theme-font (font-family-list))
      (set-frame-font theme-font t))))

(defun kwarks/disable-other-themes (&rest _)
  "Disable already-enabled themes before loading a new one."
  (mapc #'disable-theme custom-enabled-themes))

(advice-add 'load-theme :before #'kwarks/disable-other-themes)
(advice-add 'load-theme :after #'kwarks/set-theme-colors)

(defun kwarks/set-theme-faces ()
  (custom-set-faces
   `(cursor                   ((t (:background ,(kwarks/color-from-theme 100 100)))))
   '(font-lock-comment-face   ((t (:slant italic :background unspecified))))
   '(trailing-whitespace      ((t (:foreground "gray"     :background "gray"))))
   '(fringe                   ((t (:foreground "SkyBlue4" :background unspecified))))))

(defun kwarks/setup-frame (&rest args)
  "Apply titlebar theme for new frames"
  (kwarks/set-theme-font)
  (kwarks/set-titlebar-style)
  (kwarks/set-theme-colors)
  (kwarks/set-theme-faces))

(kwarks/with-make-frame (kwarks/setup-frame))
(add-hook 'after-make-frame-functions 'kwarks/setup-frame t)

(use-package doom-themes
  :init
  (setq doom-one-brighter-comments t)
  (load-theme 'doom-one t))
#+END_SRC

** Theme and geometry cache
Write the active theme's colors and frame geometry to =theme-cache.el= so
=early-init.el= can apply them before the frame is created on the next start.
This eliminates the white flash and the small-window-then-resize dance.

#+BEGIN_SRC emacs-lisp
(defun kwarks/save-theme-cache ()
  "Write current theme colors to theme-cache.el.
These are pushed into `default-frame-alist' by early-init.el before the
frame exists, so the window starts with the right colors immediately.
Frame geometry is left to desktop-restore which already handles it."
  (when (display-graphic-p)
    (let* ((cache (expand-file-name "theme-cache.el" user-emacs-directory))
           (frame (selected-frame))
           (bg (face-background 'default frame))
           (fg (face-foreground 'default frame))
           (dark-p (eq (frame-parameter frame 'background-mode) 'dark)))
      (with-temp-file cache
        (insert ";;; theme-cache.el --- auto-generated, do not edit -*- no-byte-compile: t; -*-\n")
        (when bg (insert (format "(push '(background-color . %S) default-frame-alist)\n" bg)))
        (when fg (insert (format "(push '(foreground-color . %S) default-frame-alist)\n" fg)))
        (when dark-p
          (insert "(push '(ns-appearance . dark) default-frame-alist)\n"))))))

(advice-add 'load-theme :after (lambda (&rest _) (kwarks/save-theme-cache)))
(add-hook 'kill-emacs-hook #'kwarks/save-theme-cache)
#+END_SRC
