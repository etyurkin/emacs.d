#+TITLE: denote
#+PROPERTY: header-args :tangle yes

[[https://protesilaos.com/emacs/denote][Denote]] is a simple, file-naming based note-taking system (the modern spiritual
successor to org-wiki).  Kept alongside =org-wiki= so both can be used during a
gradual migration; disable whichever you stop using.

Notes live in =denote-directory=; each file is named
=DATE--TITLE__KEYWORDS.EXT= so they are easy to find with plain tools and with
=consult= from core.

#+BEGIN_SRC emacs-lisp
(use-package denote
  :init
  (setq denote-directory (expand-file-name "denote" user-emacs-directory))
  (unless (file-directory-p denote-directory)
    (make-directory denote-directory t))
  :custom
  (denote-known-keywords '("emacs" "journal" "project" "wiki"))
  (denote-infer-keywords t)
  (denote-prompts '(title keywords))
  :hook (dired-mode . denote-dired-mode)
  :bind (("C-c n n" . denote)
         ("C-c n N" . denote-type)
         ("C-c n r" . denote-rename-file)
         ("C-c n l" . denote-link)
         ("C-c n b" . denote-backlinks)
         ("C-c n d" . denote-dired)
         ("C-c n f" . denote-open-or-create)
         ("C-c n k" . denote-rename-file-keywords)))
#+END_SRC
