;; Desktop does not restore live process buffers (like *shell*) by default.
;; Teach it to remember a shell buffer's directory and restart the shell there,
;; so a "doc on the left, shell on the right" layout comes back intact.
(defun kwarks/desktop-save-shell (_desktop-dirname)
  "Return the data needed to restore a shell buffer (its directory)."
  default-directory)

(defun kwarks/desktop-restore-shell (_file-name buffer-name dir)
  "Restart a shell named BUFFER-NAME in its saved directory DIR.
Desktop pre-creates the buffer, so set its `default-directory' explicitly;
otherwise `shell' starts the process in the buffer's stale directory."
  (let* ((dir (or dir default-directory))
         (buf (get-buffer-create buffer-name)))
    (with-current-buffer buf
      (setq default-directory dir))
    (let ((default-directory dir))
      (shell buf))
    buf))

(add-hook 'shell-mode-hook
          (lambda ()
            (setq-local desktop-save-buffer #'kwarks/desktop-save-shell)))

               '(shell-mode . kwarks/desktop-restore-shell)))

;; Don't let the desktop frameset restore frame geometry.  On macOS multi-
;; monitor/mixed-DPI setups, desktop's post-creation `set-frame-position'
;; restore drifts the frame a little on every save->restore cycle (bug#65840,
;; open since Emacs 29, unfixed in 30.x), so the size/position ends up wrong and
;; alternates between launches.  We instead save geometry to frame-geometry.el
;; on exit and apply it at frame-creation time via `default-frame-alist' in
;; early-init.el, which is deterministic.  Filtering these out of the frameset
;; means desktop still restores buffers and window splits, just not geometry.
(require 'frameset)
(dolist (param '(left top width height
                      frameset--text-pixel-width frameset--text-pixel-height))
  (setf (alist-get param frameset-filter-alist) :never))

;; Honor EMACS_NO_DESKTOP for troubleshooting: without this, the variable has
;; no effect because `desktop-read' is unconditional (see Sessions block).
(unless (getenv "EMACS_NO_DESKTOP")
  (desktop-read))

;; Persist the frame's actual geometry on exit so early-init.el can recreate it
;; next launch.  Reading it at exit captures exactly where the frame sits on
;; screen (correct, since nothing is resizing then), and writing it as
;; `default-frame-alist' pushes avoids desktop's drifting restore path entirely.
(defun kwarks/save-frame-geometry ()
  "Write the selected frame's size and position to frame-geometry.el."
  (when (display-graphic-p)
    (let* ((f      (selected-frame))
           (cache  (expand-file-name "frame-geometry.el" kwarks/cache-directory))
           (width  (frame-text-width f))
           (height (frame-text-height f))
           (left   (frame-parameter f 'left))
           (top    (frame-parameter f 'top))
           (full   (frame-parameter f 'fullscreen)))
      ;; Skip when fullscreen/maximized -- saving those pixel bounds would pin
      ;; a weird size on the next, possibly different, display.
      (unless full
        (with-temp-file cache
          (insert ";;; frame-geometry.el --- auto-generated, do not edit -*- no-byte-compile: t; lexical-binding: t; -*-\n")
          (insert (format "(push '(width . (text-pixels . %d)) default-frame-alist)\n" width))
          (insert (format "(push '(height . (text-pixels . %d)) default-frame-alist)\n" height))
          (insert (format "(push '(left . %S) default-frame-alist)\n" left))
          (insert (format "(push '(top . %S) default-frame-alist)\n" top)))))))

(add-hook 'kill-emacs-hook #'kwarks/save-frame-geometry)

Desktop restores the window layout from the previous session, which can leave
diagnostic buffers (=*Messages*=, =*Warnings*=, =*Compile-Log*=) occupying a
window on startup.  Bury them once startup finishes, keeping the rest of the
restored layout intact.
#+BEGIN_SRC emacs-lisp
(defun kwarks/hide-startup-noise-windows ()
  "Bury diagnostic buffers shown in windows left over after startup.
Switches each such window to another buffer instead of deleting it, so a
restored desktop window layout (e.g. a split) is preserved."
  (dolist (name '("*Messages*" "*Warnings*" "*Compile-Log*"))
    (let ((buf (get-buffer name)))
      (when buf
        (dolist (win (get-buffer-window-list buf nil t))
          (switch-to-prev-buffer win))))))

(add-hook 'emacs-startup-hook #'kwarks/hide-startup-noise-windows 90)

** Component file template
When following a link in =components.org= to a file that does not yet exist,
Emacs creates an empty buffer.  This hook fills it with a starter template so
the user knows the expected structure.

#+BEGIN_SRC emacs-lisp
(defun kwarks/component-template ()
  "Insert a starter template when opening a new file under components/."
  (when-let* ((file (buffer-file-name))
              (_ (string-prefix-p
                  (expand-file-name "components/" user-emacs-directory)
                  (expand-file-name file)))
              (_ (= (buffer-size) 0)))
    (let ((name (file-name-sans-extension (file-name-nondirectory file))))
      (insert "#+TITLE: " name "\n")
      (insert "#+PROPERTY: header-args :tangle yes\n\n")
      (insert ";; Describe what this component provides and any external dependencies.\n")
      (insert ";; Then replace this comment with your configuration.\n\n")
      (insert "#+BEGIN_SRC emacs-lisp\n\n#+END_SRC\n")
      (forward-line -2))))

(add-hook 'find-file-hook #'kwarks/component-template)
#+END_SRC

* ELisp Extensions
** Load elisp extensions from [lisp/ext, lisp/private]
Scripts under =lisp/ext= are loaded on demand through autoloads so they don't
cost anything at startup; the file is loaded the first time one of its commands
is invoked.  Absolute paths are used (rather than adding =lisp/ext= to
=load-path=) to avoid shadowing similarly named packages.  =lisp/private= is
still loaded eagerly.

#+BEGIN_SRC emacs-lisp
(defun load-ext (folder)
  (let ((ext-dir (expand-file-name folder user-emacs-directory)))
    (when (file-exists-p ext-dir)
      (mapc 'load (file-expand-wildcards (concat ext-dir "/*.el"))))))

(defun kwarks/autoload-ext (file &rest commands)
  "Set up autoloads for COMMANDS defined in lisp/ext/FILE."
  (let ((path (expand-file-name (concat "lisp/ext/" file) user-emacs-directory)))
    (dolist (cmd commands)
      (autoload cmd path nil t))))

(kwarks/autoload-ext "intellij-open-in-emacs" 'intellij-open-in-emacs)
(kwarks/autoload-ext "number-to-string"
                     'number-to-english-cardinal 'number-to-english-ordinal)
(kwarks/autoload-ext "org-passwords" 'kwarks/org-passwords)
(kwarks/autoload-ext "tree-mode" 'kwarks/tree)

(load-ext "lisp/private")
#+END_SRC

(defun kwarks/toggle-delete-other-windows ()
  "Toggle \\[delete-other-window] command.
If the current frame has several windows, it will act as `delete-other-windows'.
If the current frame has one window,
it will restore the window configuration to prior to full-framing."
  (interactive)

      (window-configuration-to-register window-state-register-name)

** Proxy functions
#+BEGIN_SRC emacs-lisp
(defun kwarks/set-proxy (proxy-host)
  (setq url-proxy-services
        `(("no_proxy" . "^\\(localhost\\|10.*\\)")
          ("http" . ,proxy-host)
          ("https" . ,proxy-host)))
  (setenv "http_proxy" (concat "http://" proxy-host))
  (setenv "https_proxy" (concat "http://" proxy-host))
  (setenv "HTTP_PROXY" (concat "http://" proxy-host))
  (setenv "HTTPS_PROXY" (concat "http://" proxy-host))
  (run-shell-command (concat "export http_proxy=http://" proxy-host))
  (run-shell-command (concat "export https_proxy=http://" proxy-host))
  (run-shell-command (concat "export HTTP_PROXY=http://" proxy-host))
  (run-shell-command (concat "export HTTPS_PROXY=http://" proxy-host)))

(defun proxy-on ()
  "Set proxy."
  (interactive)
  (let ((proxy-host
         (read-string "proxy host: "
                      (if (bound-and-true-p *proxy-host*) *proxy-host* ""))))
    (unless (string-blank-p proxy-host)
      (kwarks/set-proxy proxy-host)
      (message "proxy is set to %s" proxy-host))))

(defun proxy-off ()
  "Unset proxy."
  (interactive)
  (setq url-proxy-services ())
  (setenv "http_proxy" "")
  (setenv "https_proxy" "")
  (setenv "HTTP_PROXY" "")
  (setenv "HTTPS_PROXY" "")
  (run-shell-command "unset http_proxy")
  (run-shell-command "unset https_proxy")
  (run-shell-command "unset HTTP_PROXY")
  (run-shell-command "unset HTTPS_PROXY")
  (message "proxy is off"))
#+END_SRC

** XML pretty print
#+BEGIN_SRC emacs-lisp
(defun xml-pretty-print-region (begin end)
  "Pretty format XML markup in region from BEGIN to END.
You need to have 'nxml-mode'
http://www.emacswiki.org/cgi-bin/wiki/NxmlMode installed to do this.
The function inserts linebreaks to separate tags that have nothing
but whitespace between them.
It then indents the markup by using nxml's indentation rules."
  (interactive "r")
  (save-excursion
    (nxml-mode)
    (goto-char begin)
    (while (search-forward-regexp "\>[ \\t]*\<" nil t)
      (backward-char) (insert "\n") (setq end (1+ end)))
    (indent-region begin end))
  (message "Ah, much better!"))

** Show decrypted org entry
org-decrypt-entry modifies org file by replacing encrypted content.
This function decrypts and displays org-mode encrypted entry in a separate read-only buffer.

#+BEGIN_SRC emacs-lisp
(defun kwarks/org-display-decrypted-entry ()
  "Display decrypted content of the current headline in a new read-only buffer."
  (interactive)
  (require 'epg)
  (unless (org-before-first-heading-p)
    (org-with-wide-buffer
     (org-back-to-heading t)
     (let ((heading-point (point))
	   (heading-was-invisible-p
	    (save-excursion
	      (outline-end-of-heading)
	      (org-invisible-p))))
       (org-end-of-meta-data)
       (when (looking-at "-----BEGIN PGP MESSAGE-----")
	 (setq-local epg-context (epg-make-context nil t t))
	 (let* ((end (save-excursion
		       (search-forward "-----END PGP MESSAGE-----")
		       (forward-line)
		       (point)))
		(encrypted-text (buffer-substring-no-properties (point) end))
		(decrypted-text
		 (decode-coding-string
		  (epg-decrypt-string
		   epg-context
		   encrypted-text)
		  'utf-8)))
	   (let* ((entry-name (elt (org-heading-components) 4))
		  (buf-name (concat "*decrypted: " entry-name  "*"))
		  (buf (get-buffer-create buf-name)))
	     (with-current-buffer buf
	       (read-only-mode -1)
	       (erase-buffer)
	       (insert decrypted-text)
	       (read-only-mode)
	       (org-mode)
	       (pop-to-buffer buf)))
	   nil))))))
#+END_SRC

** Toggle image display
#+BEGIN_SRC emacs-lisp
(defvar-local endless/display-images t)

(defun endless/toggle-image-display ()
  "Toggle images display on current buffer."
  (interactive)
  (setq endless/display-images
        (null endless/display-images))
  (endless/backup-display-property endless/display-images))

(defun endless/backup-display-property (invert &optional object)
  "Move the 'display property at POS to 'display-backup.
Only applies if display property is an image.
If INVERT is non-nil, move from 'display-backup to 'display
instead.
Optional OBJECT specifies the string or buffer. Nil means current
buffer."
  (let* ((inhibit-read-only t)
         (from (if invert 'display-backup 'display))
         (to (if invert 'display 'display-backup))
         (pos (point-min))
         left prop)
    (while (and pos (/= pos (point-max)))
      (if (get-text-property pos from object)
          (setq left pos)
        (setq left (next-single-property-change pos from object)))
      (if (or (null left) (= left (point-max)))
          (setq pos nil)
        (setq prop (get-text-property left from object))
        (setq pos (or (next-single-property-change left from object)
                      (point-max)))
        (when (eq (car prop) 'image)
          (add-text-properties left pos (list from nil to prop) object))))))

(defun server-shutdown ()
  "Save buffers and the desktop, then shut down Emacs.
`kill-emacs' is supposed to save the desktop via `kill-emacs-hook', but that
path does not reliably fire here, so save the desktop explicitly first."
  (interactive)

  ;; the kill-emacs-hook does not reliably fire on this path.
  (when (fboundp 'kwarks/save-frame-geometry)
    (kwarks/save-frame-geometry))
  (when (and (bound-and-true-p desktop-save-mode)
             desktop-dirname)
    (desktop-save-in-desktop-dir))
  (kill-emacs))

(defun kwarks/sql-conect (product database-name)
  (if (bound-and-true-p secrets-file)
      (require 'secrets (expand-file-name secrets-file)))

  ;; update the password to the sql-connection-alist
  (let ((connection-info (assoc database-name sql-connection-alist))
        (sql-password (car (last (assoc database-name sql-passwords)))))
    (delete sql-password connection-info)

    (setq sql-connection-alist (assq-delete-all database-name sql-connection-alist))
    (add-to-list 'sql-connection-alist connection-info))

  ;; connect to database
  (setq sql-product product)
  (sql-connect database-name))
